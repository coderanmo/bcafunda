#Looping in python
#In python, we can use else condition statement
with while loop
#sum of n natural numbers
n=int(input("Enter n:"))
s=0
i=1
while (i<=n):
s=s+i
i=i+1
#else condition executes only when the whole
while loop runs
#if any break occurs in-between then else
doesn't get executed
else:
print("sum of ",n,":",s)
