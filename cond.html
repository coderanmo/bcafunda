#Program to find out largest of three numbers using conditional statements
#Taking input of three integer numbers
a = int(input("Enter a: "))
b = int(input("Enter b: "))
c = int(input("Enter c: "))
#conditional statements
if a>b and a>c:
print("a is largest having value ",a)
elif b>a and b>c:
print("b is largest having value ",b)
else:
print("c is largest having value ",c)
